#include<stdio.h>
int main(){
    char i;
    int a,b;
    scanf("%d%c%d",&a,&i,&b);
    switch(i){
        case'+':
        printf("result = %d",a+b);
        break;
        case '-':
        printf("result = %d",a-b);
        break;
        case '*':
        printf("result = %d",a*b);
        break;
        case '/':
        printf("result = %d",a/b);
        break;
        default:
        printf("enter valid symbol");
        
    }
    
    return 0;
}
